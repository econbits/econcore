load("encoding", "json_decode", "json_encode")

jstr = '{"value":"v"}'
j = json_decode(jstr)

assert(len(j) == 1, msg=str(j))
assert(j["value"] == "v", msg=str(j))

ret = json_encode(j)
assert(jstr == ret, msg=jstr + " != " + ret)
